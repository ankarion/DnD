table without id
sum(number(filter(flat(rows.rewards), (t)=>contains(t, "оп")))) as оп,
sum(number(filter(flat(rows.rewards), (t)=>contains(t, "зм")))) as зм,
sum(number(filter(flat(rows.rewards), (t)=>contains(t, "сухпай")))) as сухпай
